CREATE TABLE EMPLOYEE(
    E_ID NUMBER(3),
    ENAME VARCHAR2(21),
    AGE NUMBER(2),
    SALARY NUMBER(7,2),
    CONSTRAINT EPKEY PRIMARY KEY(E_ID)
);

SET SERVEROUTPUT ON;

--CREATING OF CURSOR
DECLARE
V_E_ID EMPLOYEE.E_ID % TYPE;
V_E_NAME EMPLOYEE.ENAME % TYPE;
V_E_AGE EMPLOYEE.AGE % TYPE;
V_E_SAL EMPLOYEE.SALARY % TYPE;

CURSOR EMP_CURSOR IS SELECT * FROM EMPLOYEE;

BEGIN
	OPEN EMP_CURSOR;
	LOOP
		FETCH EMP_CURSOR INTO V_E_ID,V_E_NAME,V_E_AGE,V_E_SAL;
		EXIT WHEN EMP_CURSOR % NOTFOUND;
		DBMS_OUTPUT.PUT_LINE('EMPLOYEE ID : '|| V_E_ID);
		DBMS_OUTPUT.PUT_LINE('EMPLOYEE NAME : '|| V_E_NAME);
		DBMS_OUTPUT.PUT_LINE('EMPLOYEE AGE : '|| V_E_AGE);
		DBMS_OUTPUT.PUT_LINE('EMPLOYEE SALARY : '|| V_E_SAL);
		DBMS_OUTPUT.PUT_LINE('-------------------------------------------------------------------');
	END LOOP;
	CLOSE EMP_CURSOR;
END;
